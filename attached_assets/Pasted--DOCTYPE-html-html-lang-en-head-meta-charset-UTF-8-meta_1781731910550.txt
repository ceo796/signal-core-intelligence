<body>

  <!-- Optional live controls for development (remove for production) -->
  <div class="controls">
    <label>
      Speed
      <input type="range" id="speed" min="0.2" max="2.5" step="0.1" value="1.0">
    </label>
    <label>
      Attraction
      <input type="range" id="attraction" min="0" max="0.8" step="0.05" value="0.35">
    </label>

  <script>
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d', { alpha: true });

    let width, height, dpr;
    let orbs = [];
    let mouse = { x: 0, y: 0, active: false };
    let animationFrame;
    let reducedMotion = false;

    // Configuration (easy to tweak)
    const config = {
      orbCount: 16,
      baseSpeed: 0.35,
      attractionStrength: 0.35,
      showLines: true,
      maxLineDistance: 180
    };

    // Color palette - subtle professional (white-friendly)
    const palette = [
      { color: '#0f172a', alpha: 0.12 },   // Deep navy
      { color: '#334155', alpha: 0.10 },   // Slate
      { color: '#0ea5e9', alpha: 0.08 },   // Sky blue accent
      { color: '#64748b', alpha: 0.09 }    // Muted gray
    ];

    function resize() {
      dpr = Math.min(window.devicePixelRatio || 1, 2);
      width = window.innerWidth;
      height = window.innerHeight;

      canvas.width = width * dpr;
      canvas.height = height * dpr;
      canvas.style.width = width + 'px';
      canvas.style.height = height + 'px';
      ctx.scale(dpr, dpr);
    }

    class Orb {
      constructor() {
        this.reset();
        this.x = Math.random() * width;
        this.y = Math.random() * height;
      }

      reset() {
        this.x = Math.random() * width;
        this.y = Math.random() * height;
        this.size = Math.random() * 3.5 + 1.8;
        this.depth = Math.random() * 0.6 + 0.4; // Parallax factor
        this.vx = (Math.random() - 0.5) * 0.6;
        this.vy = (Math.random() - 0.5) * 0.6;
        this.hue = palette[Math.floor(Math.random() * palette.length)];
      }

      update() {
        // Gentle autonomous drift
        this.x += this.vx * config.baseSpeed;
        this.y += this.vy * config.baseSpeed;

        // Very soft center bias (keeps composition cohesive)
        const centerX = width / 2;
        const centerY = height / 2;
        this.vx += (centerX - this.x) * 0.000008;
        this.vy += (centerY - this.y) * 0.000008;

        // Mouse interaction (attraction)
        if (mouse.active) {
          const dx = mouse.x - this.x;
          const dy = mouse.y - this.y;
          const dist = Math.sqrt(dx * dx + dy * dy) || 1;
          const force = (config.attractionStrength * this.depth) / (dist * 0.008 + 1);

          this.vx += dx * force * 0.012;
          this.vy += dy * force * 0.012;
        }

        // Damping for silky motion
        this.vx *= 0.982;
        this.vy *= 0.982;

        // Soft bounds
        if (this.x < -50) this.x = width + 50;
        if (this.x > width + 50) this.x = -50;
        if (this.y < -50) this.y = height + 50;
        if (this.y > height + 50) this.y = -50;
      }

      draw() {
        ctx.save();
        ctx.globalAlpha = this.hue.alpha;
        ctx.fillStyle = this.hue.color;

        // Soft glow
        ctx.shadowColor = this.hue.color;
        ctx.shadowBlur = this.size * 2.5;

        ctx.beginPath();
        ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
        ctx.fill();

        ctx.shadowBlur = 0;
        ctx.restore();
      }
    }

    function initOrbs() {
      orbs = [];
      for (let i = 0; i < config.orbCount; i++) {
        orbs.push(new Orb());
      }
    }

    function drawConnections() {
      if (!config.showLines) return;

      ctx.strokeStyle = '#0f172a';
      ctx.lineWidth = 0.6;

      for (let i = 0; i < orbs.length; i++) {
        for (let j = i + 1; j < orbs.length; j++) {
          const dx = orbs[i].x - orbs[j].x;
          const dy = orbs[i].y - orbs[j].y;
          const dist = Math.sqrt(dx * dx + dy * dy);

          if (dist < config.maxLineDistance) {
            const alpha = (1 - dist / config.maxLineDistance) * 0.18;
            ctx.globalAlpha = alpha;
            ctx.beginPath();
            ctx.moveTo(orbs[i].x, orbs[i].y);
            ctx.lineTo(orbs[j].x, orbs[j].y);
            ctx.stroke();
          }
        }
      }
      ctx.globalAlpha = 1;
    }

    function animate() {
      ctx.clearRect(0, 0, width, height);

      // Update + draw orbs
      for (const orb of orbs) {
        orb.update();
        orb.draw();
      }

      drawConnections();

      animationFrame = requestAnimationFrame(animate);
    }

    function setupInteraction() {
      const handleMove = (e) => {
        const rect = canvas.getBoundingClientRect();
        mouse.x = e.clientX - rect.left;
        mouse.y = e.clientY - rect.top;
        mouse.active = true;
      };

      const handleLeave = () => {
        mouse.active = false;
      };

      canvas.addEventListener('mousemove', handleMove);
      canvas.addEventListener('mouseleave', handleLeave);

      // Touch support (lighter interaction on mobile)
      canvas.addEventListener('touchmove', (e) => {
        e.preventDefault();
        const rect = canvas.getBoundingClientRect();
        mouse.x = e.touches[0].clientX - rect.left;
        mouse.y = e.touches[0].clientY - rect.top;
        mouse.active = true;
      }, { passive: false });

      canvas.addEventListener('touchend', handleLeave);

      // Reduced motion preference
      const mediaQuery = window.matchMedia('(prefers-reduced-motion: reduce)');
      reducedMotion = mediaQuery.matches;

      mediaQuery.addEventListener('change', (e) => {
        reducedMotion = e.matches;
        if (reducedMotion) {
          cancelAnimationFrame(animationFrame);
        } else {
          animate();
        }
      });
    }

    function setupControls() {
      const speedSlider = document.getElementById('speed');
      const attractionSlider = document.getElementById('attraction');
      const linesToggle = document.getElementById('lines');

      speedSlider.addEventListener('input', (e) => {
        config.baseSpeed = parseFloat(e.target.value);
      });

      attractionSlider.addEventListener('input', (e) => {
        config.attractionStrength = parseFloat(e.target.value);
      });

      linesToggle.addEventListener('change', (e) => {
        config.showLines = e.target.checked;
      });

      // Keyboard accessibility
      document.addEventListener('keydown', (e) => {
        if (e.key === '?') {
          document.querySelector('.controls').style.display =
            document.querySelector('.controls').style.display === 'none' ? 'flex' : 'none';
        }
      });
    }

    function init() {
      resize();
      initOrbs();
      setupInteraction();
      setupControls();

      // Start animation unless user prefers reduced motion
      if (!reducedMotion) {
        animate();
      } else {
        // Draw static elegant state
        ctx.clearRect(0, 0, width, height);
        for (const orb of orbs) {
          orb.draw();
        }
      }

      // Handle resize
      let resizeTimeout;
      window.addEventListener('resize', () => {
        clearTimeout(resizeTimeout);
        resizeTimeout = setTimeout(() => {
          resize();
          initOrbs(); // Re-seed positions on significant resize
        }, 150);
      });

      // Pause when tab is hidden (performance)
      document.addEventListener('visibilitychange', () => {
        if (document.hidden) {
          cancelAnimationFrame(animationFrame);
        } else if (!reducedMotion) {
          animate();
        }
      });

      // Easter egg: click to respawn orbs
      canvas.addEventListener('click', () => {
        initOrbs();
      });
    }

    // Boot everything
    init();
  </script>
