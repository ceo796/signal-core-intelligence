import React, { useEffect, useRef } from 'react';

export const GridWave: React.FC = () => {
  const canvasRef = useRef<HTMLCanvasElement | null>(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    let animationFrameId: number;
    let phase = 0;

    // Handle resizing
    const resizeCanvas = () => {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight * 0.6; // Limits it to the top portion of the page
    };
    window.addEventListener('resize', resizeCanvas);
    resizeCanvas();

    // Animation Loop
    const draw = () => {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      
      // Grid configuration
      const rows = 25;
      const cols = 40;
      const cellWidth = canvas.width / cols;
      const cellHeight = canvas.height / rows;
      
      phase += 0.02; // Controls the speed of the wave

      ctx.beginPath();
      // Set grid line styling (subtle blue/gray to match Precision Document Intelligence theme)
      ctx.strokeStyle = 'rgba(59, 130, 246, 0.15)'; 
      ctx.lineWidth = 1;

      for (let r = 0; r < rows; r++) {
        for (let c = 0; c < cols; c++) {
          // Calculate a wave offset based on grid position and time (phase)
          const waveX = Math.sin(c * 0.15 + phase) * 15;
          const waveY = Math.cos(r * 0.15 + phase) * 15;

          // 3D Perspective math (compressing the top to look like it recedes into the distance)
          const perspectiveFactor = (r / rows); // 0 at top, 1 at bottom
          
          const x = c * cellWidth + (waveX * perspectiveFactor);
          // Standardize screen placement with perspective compressing the top lines
          const y = (r * cellHeight * perspectiveFactor) + waveY;

          // Drawing horizontal grid lines
          if (c < cols - 1) {
            const nextX = (c + 1) * cellWidth + (Math.sin((c + 1) * 0.15 + phase) * 15 * perspectiveFactor);
            ctx.moveTo(x, y);
            ctx.lineTo(nextX, y);
          }

          // Drawing vertical grid lines
          if (r < rows - 1) {
            const nextPerspective = ((r + 1) / rows);
            const nextY = ((r + 1) * cellHeight * nextPerspective) + Math.cos((r + 1) * 0.15 + phase) * 15;
            const nextX = c * cellWidth + (Math.sin(c * 0.15 + phase) * 15 * nextPerspective);
            ctx.moveTo(x, y);
            ctx.lineTo(nextX, nextY);
          }
        }
      }
      ctx.stroke();

      animationFrameId = requestAnimationFrame(draw);
    };

    draw();

    return () => {
      window.removeEventListener('resize', resizeCanvas);
      cancelAnimationFrame(animationFrameId);
    };
  }, []);

  return (
    <canvas
      ref={canvasRef}
      className="absolute top-0 left-0 w-full pointer-events-none z-0"
      style={{ maskImage: 'linear-gradient(to bottom, rgba(0,0,0,1) 70%, rgba(0,0,0,0))' }}
    />
  );
};
